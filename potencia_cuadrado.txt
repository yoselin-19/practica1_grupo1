public int potencia_Cuadrado (double numero) {
	double resultado = Math.pow(2, numero);
	return resultado;
}

public double raiz_cuadrada (double numero) {
	double resultado = Math.sqrt(numero);
	return resultado;
}

public double potencia_cubo(double numero) {
	 double valor = numero * numero * numero;
   return valor;
}

public double raiz_cubica(double numero) {
	 double valor = Math.pow(numero, 1/3); 
   return valor;
}

public double multiplicacion(double num1, double num2) {
	 double valor = num1 * num2; 
   return valor;
}

public double division(double num1, double num2) {
	 double valor = num1 / num2; 
   return valor;
}
